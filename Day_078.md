6. *Verify the Log of Node added with keyword "Agent successfully connected and online"*

7. *Create a new job with pipeline as option -> Name it -> Build steps -> Pipeline script*
```
pipeline {
    agent { label 'ststor01' }
    
    parameters {
        string(name: 'BRANCH', defaultValue: 'master', description: 'parameter for branch')
    }
    stages {
        stage('Deploy') {
            steps {
                script {
                    def branch = params.BRANCH
                    echo "Using ${branch}"
              
                    if (params.BRANCH == 'master') {
                        echo "Deploying master branch..."
                        sh 'rm -rf /tmp/web_app'
                        sh 'git clone -b master http://git.stratos.xfusioncorp.com/sarah/web_app.git /tmp/web_app'
                        sh 'ls -la /tmp/web_app'
                        sh 'echo "Bl@kW" | sudo -S cp -r /tmp/web_app/* /var/www/html/'
                        sh 'rm -rf /tmp/web_app'
                    } 
                
                    else if (params.BRANCH == 'feature') {
                        echo "Deploying feature branch..."
                        sh 'rm -rf /tmp/web_app'
                        sh 'git clone -b master http://git.stratos.xfusioncorp.com/sarah/web_app.git /tmp/web_app'
                        sh 'ls -la /tmp/web_app'
                        sh 'echo "Bl@kW" | sudo -S cp -r /tmp/web_app/* /var/www/html/'
                        sh 'rm -rf /tmp/web_app'
                    } 
                    
                    else {
                        error "Invalid BRANCH parameter value: ${params.BRANCH}"                      
                    }
                }
            }
        }
    }
}
```

8. *Build the Job*
